### Phase 3: Rate Limiting (Day 3)

**Task:** Add rate limiting to AI endpoints

```typescript
// lib/rate-limiter.ts - NEW FILE
import { NextRequest, NextResponse } from 'next/server';

interface RateLimitConfig {
  windowMs: number;
  maxRequests: number;
  message?: string;
}

const requests = new Map<string, number[]>();

export function rateLimit(config: RateLimitConfig) {
  return async (req: NextRequest, userId: string): Promise<NextResponse | null> => {
    const now = Date.now();
    const windowStart = now - config.windowMs;
    
    // Get user's request history
    const userRequests = requests.get(userId) || [];
    
    // Filter to current window
    const recentRequests = userRequests.filter(time => time > windowStart);
    
    // Check limit
    if (recentRequests.length >= config.maxRequests) {
      return NextResponse.json(
        { 
          error: config.message || 'Too many requests, please slow down',
          retryAfter: Math.ceil((recentRequests[0] - windowStart) / 1000)
        },
        { status: 429 }
      );
    }
    
    // Record request
    recentRequests.push(now);
    requests.set(userId, recentRequests);
    
    return null; // Allow request
  };
}

// Predefined limiters
export const aiRateLimit = rateLimit({
  windowMs: 60 * 1000,  // 1 minute
  maxRequests: 20,
  message: 'Too many AI requests. Please wait a moment.'
});

export const strictRateLimit = rateLimit({
  windowMs: 60 * 1000,
  maxRequests: 5,
  message: 'This action is rate limited. Please wait.'
});
```

**Apply to chat route:**

### Phase 4: Error Handling Utility (Day 3-4)

**Task:** Create standardized error responses

```typescript
// lib/api/responses.ts - NEW FILE
import { NextResponse } from 'next/server';

interface ApiError {
  message: string;
  code?: string;
  details?: any;
}

interface ApiSuccess<T> {
  data: T;
  message?: string;
}

// Standard error responses
export const apiError = {
  unauthorized: (message = 'Unauthorized') => 
    NextResponse.json({ error: message, code: 'UNAUTHORIZED' }, { status: 401 }),
  
  forbidden: (message = 'Forbidden') =>
    NextResponse.json({ error: message, code: 'FORBIDDEN' }, { status: 403 }),
  
  notFound: (resource = 'Resource') =>
    NextResponse.json({ error: `${resource} not found`, code: 'NOT_FOUND' }, { status: 404 }),
  
  badRequest: (message: string, details?: any) =>
    NextResponse.json({ error: message, code: 'BAD_REQUEST', details }, { status: 400 }),
  
  internal: (message = 'Internal server error') =>
    NextResponse.json({ error: message, code: 'INTERNAL_ERROR' }, { status: 500 }),
  
  rateLimited: (retryAfter?: number) =>
    NextResponse.json(
      { error: 'Too many requests', code: 'RATE_LIMITED', retryAfter },
      { status: 429 }
    ),
  
  custom: (message: string, status: number, code?: string, details?: any) =>
    NextResponse.json({ error: message, code, details }, { status }),
};

// Standard success responses
export const apiSuccess = {
  ok: <T>(data: T, message?: string) =>
    NextResponse.json({ data, message }, { status: 200 }),
  
  created: <T>(data: T, message?: string) =>
    NextResponse.json({ data, message }, { status: 201 }),
  
  noContent: () =>
    new NextResponse(null, { status: 204 }),
  
  custom: <T>(data: T, status: number, message?: string) =>
    NextResponse.json({ data, message }, { status }),
};
```

**Usage example:**
```typescript
// app/api/conversations/route.ts
import { apiError, apiSuccess } from '@/lib/api/responses';

export async function GET(request: NextRequest) {
  const { userId } = await auth();
  
  if (!userId) {
    return apiError.unauthorized();
  }
  
  try {
    const conversations = await prisma.conversation.findMany({
      where: { userId }
    });
    
    return apiSuccess.ok({ conversations });
  } catch (error) {
    console.error('Failed to fetch conversations:', error);
    return apiError.internal('Failed to fetch conversations');
  }
}
```

```typescript
export const ChatMessageSchema = z.object({
  role: z.enum(['user', 'assistant']),
  content: z.string().min(1).max(10000),
  fileAttachments: z.array(z.object({
    name: z.string(),
    url: z.string().url(),
    type: z.string(),

  messages: z.array(ChatMessageSchema).min(1).max(50),
  conversationId: z.string().optional(),
  userId: z.string().optional(),
```
